def catigorize_col(df,col,labels):
    edges=[df[col].describe()["min"],
           df[col].describe()["25%"],
           df[col].describe()["50%"],
           df[col].describe()["75%"],
          df[col].describe()["max"]]
    df[col]=pd.cut(df[col],edges,labels=labels,duplicates="drop")
    return df
# pd.cut function--> numerical value ko ranges mai divide krke categorical values m convert krta h like avg,max,min
